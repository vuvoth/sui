4//! BigTable Store implementation for sui-indexer-alt-framework.
5//!
6//! This implements the `Store` and `Connection` traits to allow the new framework
7//! to use BigTable for watermark storage. The watermark is stored in the same format
8//! as the legacy binary (key=[0] in watermark_alt table) to allow both binaries to
9//! run in parallel during migration.
10//!
11//! Note: The legacy binary stores "next checkpoint to process", while the new framework
12//! uses "checkpoint_hi_inclusive" (last processed). We translate between these with +1/-1.
13

26/// A Store implementation backed by BigTable.
27///
28/// This store manages watermarks in the `watermark_alt` table using key `[0]`
29/// for compatibility with the legacy kvstore binary.
30#[derive(Clone)]
31pub struct BigTableStore {
32    client: BigTableClient,
33}
34
35impl BigTableStore {
36    pub fn new(client: BigTableClient) -> Self {
37        Self { client }
38    }
39}

41#[async_trait]
42impl Store for BigTableStore {
43    type Connection<'c> = BigTableConnection<'c>;
44
45    async fn connect<'c>(&'c self) -> Result<Self::Connection<'c>> {
46        Ok(BigTableConnection {
47            client: self.client.clone(),
48            _marker: std::marker::PhantomData,
49        })
50    }
51}

53/// A connection to BigTable for watermark operations and data writes.
54pub struct BigTableConnection<'a> {
55    client: BigTableClient,
56    _marker: std::marker::PhantomData<&'a ()>,
57}
58
59impl BigTableConnection<'_> {
60    /// Process a checkpoint by writing all its data to BigTable.
61    ///
62    /// This is a copy of KvWorker::process_checkpoint to decouple the new binary
63    /// from the legacy code path. Changes here won't affect the legacy binary.
64    pub async fn process_checkpoint(&mut self, checkpoint: &CheckpointData) -> Result<()> {
65        let mut objects = vec![];
66        let mut transactions = vec![];
67        for transaction in &checkpoint.transactions {
68            let full_transaction = TransactionData {
69                transaction: transaction.transaction.clone(),
70                effects: transaction.effects.clone(),
71                events: transaction.events.clone(),
72                checkpoint_number: checkpoint.checkpoint_summary.sequence_number,
73                timestamp: checkpoint.checkpoint_summary.timestamp_ms,
74            };
75            for object in &transaction.output_objects {
76                objects.push(object);
77            }
78            transactions.push(full_transaction);
79        }
80        self.client
81            .save_objects(&objects, checkpoint.checkpoint_summary.timestamp_ms)
82            .await?;
83        self.client.save_transactions(&transactions).await?;
84        self.client.save_checkpoint(checkpoint).await?;
85        if let Some(epoch_info) = checkpoint.epoch_info()? {
86            if epoch_info.epoch > 0 {
87                let mut prev = self
88                    .client
89                    .get_epoch(epoch_info.epoch - 1)
90                    .await?
91                    .with_context(|| {
92                        format!(
93                            "previous epoch {} not found when processing epoch {}",
94                            epoch_info.epoch - 1,
95                            epoch_info.epoch
96                        )
97                    })?;
98                prev.end_checkpoint = epoch_info.start_checkpoint.map(|sq| sq - 1);
99                prev.end_timestamp_ms = epoch_info.start_timestamp_ms;
100                self.client.save_epoch(prev).await?;
101            }
102            self.client.save_epoch(epoch_info).await?;
103        }
104        Ok(())
105    }
106}

108#[async_trait]
109impl Connection for BigTableConnection<'_> {
110    async fn init_watermark(
111        &mut self,
112        _pipeline_task: &str,
113        _default_next_checkpoint: u64,
114    ) -> Result<Option<u64>> {
115        // Read existing watermark from BigTable.
116        // The stored value is "next checkpoint to process", so we subtract 1
117        // to get checkpoint_hi_inclusive (last processed checkpoint).
118        //
119        // Phase 1: We ignore pipeline_task and use the shared key [0] for
120        // compatibility with the legacy binary.
121        let next_checkpoint = self.client.get_latest_checkpoint().await?;
122        if next_checkpoint == 0 {
123            // No watermark exists yet - return None to indicate initialization.
124            // We don't actually write the default because it would be incompatible
125            // with the legacy binary.
126            Ok(None)
127        } else {
128            // Return the last processed checkpoint (next - 1)
129            Ok(Some(next_checkpoint - 1))
130        }
131    }
132
133    async fn committer_watermark(
134        &mut self,
135        _pipeline_task: &str,
136    ) -> Result<Option<CommitterWatermark>> {
137        let next_checkpoint = self.client.get_latest_checkpoint().await?;
138        if next_checkpoint == 0 {
139            Ok(None)
140        } else {
141            Ok(Some(CommitterWatermark {
142                epoch_hi_inclusive: 0,
143                checkpoint_hi_inclusive: next_checkpoint - 1,
144                tx_hi: 0,
145                timestamp_ms_hi_inclusive: 0,
146            }))
147        }
148    }
149
150    async fn set_committer_watermark(
151        &mut self,
152        _pipeline_task: &str,
153        watermark: CommitterWatermark,
154    ) -> Result<bool> {
155        // Write checkpoint number to watermark_alt table.
156        // The legacy binary stores "next checkpoint to process", so we add 1
157        // to checkpoint_hi_inclusive.
158        // Uses checkpoint timestamp for cell timestamp (deterministic).
159        self.client
160            .save_watermark(watermark.checkpoint_hi_inclusive + 1)
161            .await?;
162        Ok(true)
163    }
164
165    // Phase 1: Return Ok(None) - reader/pruner watermarks not needed for concurrent
166    // pipelines without pruning.
167
168    async fn reader_watermark(
169        &mut self,
170        _pipeline: &'static str,
171    ) -> Result<Option<ReaderWatermark>> {
172        Ok(None)
173    }
174
175    async fn pruner_watermark(
176        &mut self,
177        _pipeline: &'static str,
178        _delay: Duration,
179    ) -> Result<Option<PrunerWatermark>> {
180        Ok(None)
181    }
182
183    async fn set_reader_watermark(
184        &mut self,
185        _pipeline: &'static str,
186        _reader_lo: u64,
187    ) -> Result<bool> {
188        Ok(false)
189    }
190
191    async fn set_pruner_watermark(
192        &mut self,
193        _pipeline: &'static str,
194        _pruner_hi: u64,
195    ) -> Result<bool> {
196        Ok(false)
197    }
198}
