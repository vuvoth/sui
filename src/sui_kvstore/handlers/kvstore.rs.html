4use std::sync::Arc;
5
6use sui_indexer_alt_framework::pipeline::Processor;
7use sui_indexer_alt_framework::pipeline::concurrent::{BatchStatus, Handler};
8use sui_indexer_alt_framework_store_traits::Store;
9use sui_types::full_checkpoint_content::{Checkpoint, CheckpointData};
10
11use crate::bigtable::store::BigTableStore;
12
13/// Pipeline that writes checkpoint data to BigTable.
14/// Mirrors the behavior of the legacy KvWorker.
15pub struct KvStorePipeline;
16
17#[async_trait::async_trait]
18impl Processor for KvStorePipeline {
19    const NAME: &'static str = "kvstore";
20    type Value = CheckpointData;
21
22    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
23        // Convert from new Checkpoint type to legacy CheckpointData
24        Ok(vec![checkpoint.as_ref().clone().into()])
25    }
26}
27
28#[async_trait::async_trait]
29impl Handler for KvStorePipeline {
30    type Store = BigTableStore;
31    type Batch = Option<CheckpointData>;
32
33    fn batch(
34        &self,
35        batch: &mut Self::Batch,
36        values: &mut std::vec::IntoIter<Self::Value>,
37    ) -> BatchStatus {
38        // Take one checkpoint at a time. Combined with write_concurrency,
39        // this allows multiple checkpoints in flight while ensuring retries
40        // only affect a single checkpoint.
41        if batch.is_none() && values.len() > 0 {
42            *batch = values.next();
43            BatchStatus::Ready
44        } else {
45            BatchStatus::Pending
46        }
47    }
48
49    async fn commit<'a>(
50        &self,
51        batch: &Self::Batch,
52        conn: &mut <Self::Store as Store>::Connection<'a>,
53    ) -> anyhow::Result<usize> {
54        if let Some(checkpoint) = batch {
55            conn.process_checkpoint(checkpoint).await?;
56            Ok(1)
57        } else {
58            Ok(0)
59        }
60    }
61}
